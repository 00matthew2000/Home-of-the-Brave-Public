	id = worldnews.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event worldnews.1" }

	desc = {
		text = worldnews.1.d.all
		trigger = {
			country_exists = CSA

			country_exists = CAL
		}
	}
	desc = {
		text = worldnews.1.d.nocal
		trigger = {
			country_exists = CSA
			country_exists = TEX
			NOT = { country_exists = CAL }
		}
	}
	desc = {
		text = worldnews.1.d.csa
		trigger = {
			country_exists = CSA
			NOT = { country_exists = TEX }
			NOT = { country_exists = CAL }
		}
	}
	desc = {
		text = worldnews.1.d.aus
		trigger = {
			NOT = { country_exists = CSA }
			country_exists = TEX
			NOT = { country_exists = CAL }
		}
	}

	major = yes
	is_triggered_only = yes

	trigger = {
		wants_world_news = yes
	}

	option = {
		name = worldnews.1.a
	}
}
